  <script>
    let _data = null;
    let _currentSpecificID = null;

    window.onload = load;
    function load() { google.script.run.withSuccessHandler(render).getSidebarData(); }

    function render(d) {
      _data = d;
      // Tab 1
      const sel = document.getElementById('impName');
      sel.innerHTML = d.importList.map(t=>`<option value="${t.name}">${t.name}</option>`).join('');
      fillCode();
      // Tab 2 & 4 (Tree)
      renderTree(d.inventoryTree, 'treeReport', 'checkbox');
      renderTree(d.inventoryTree, 'treeBroken', 'radio');
      // Tab 3
      const listExp = document.getElementById('listExp');
      if(d.expiredTools.length===0) listExp.innerHTML='<i style="color:#999">Không có</i>';
      else {
        listExp.innerHTML=d.expiredTools.map(t=>`
          <div class="card expired" id="card-${t.id}">
            <div class="card-info"><b>${t.name}</b><small>${t.id} (${t.used}/${t.limit})</small></div>
            <div style="display:flex; gap:5px">
              <button class="btn btn-secondary btn-sm" onclick="quickReport('${t.id}')" title="Báo cáo"><i class="fas fa-file-invoice"></i></button>
              <button class="btn btn-secondary btn-sm" onclick="manualFinish('${t.id}')" title="Đã xong (Xóa)"><i class="fas fa-check"></i></button>
            </div>
          </div>`).join('');
      }
      const listNear = document.getElementById('listNear');
      listNear.innerHTML=d.aboutToExpire.length?d.aboutToExpire.map(t=>`<div class="card near"><div class="card-info"><b>${t.name}</b><small>Còn ${t.remaining}</small></div></div>`).join(''):'<i style="color:#999">Không có</i>';
    }

    // -- Logic Tree View --
    function renderTree(tree, cid, type) {
      let html = "";
      const nameAttr = cid === 'treeBroken' ? 'name="broken_select"' : '';
      let hasItem = false;
      for (const grp in tree) {
         // Lọc riêng cho Tab báo hỏng: Nếu group không còn item nào (do đã hết/hỏng hết) thì không hiện group đó
         const items = tree[grp];
         // Với brokenList, getSidebarData đã lọc các item used >= limit rồi.
         // Nhưng tree là lấy full từ getSidebarData, ta cần check lại logic hiển thị.
         // Tuy nhiên, để đơn giản, treeReport hiển thị tất cả, treeBroken cần hiển thị những cái còn dùng được.
         // => Ta sẽ dùng logic filter tại frontend dựa trên danh sách 'brokenList' từ backend gửi về cho chính xác.
         
         let displayItems = items;
         if(cid === 'treeBroken') {
             // Chỉ lấy những item có ID nằm trong brokenList
             displayItems = items.filter(i => _data.brokenList.find(b => b.id === i.id));
         }

         if(displayItems.length > 0) {
            hasItem = true;
            const grpInput = type==='checkbox' ? `<input type="checkbox" class="tree-cb grp-cb" onchange="toggleGroup('${grp}',this.checked,'${cid}')" onclick="event.stopPropagation()">` : '';
            html += `<details id="${cid}-grp-${grp}"><summary>${grpInput}${grp}</summary><ul class="sub-list">${displayItems.map(i=>`<li class="sub-item"><input type="${type}" ${nameAttr} class="tree-cb item-cb ${cid}-item-${grp}" value="${i.id}" onchange="checkParent('${grp}','${cid}')"><label onclick="this.previousElementSibling.click()">${i.id} (${i.used}/${i.limit})</label></li>`).join('')}</ul></details>`;
         }
      }
      if(!hasItem && cid === 'treeBroken') html = '<i style="color:#999">Không có vật tư nào để báo hỏng.</i>';
      document.getElementById(cid).innerHTML = html;
    }

    function toggleGroup(name, checked, cid) { document.querySelectorAll(`.${cid}-item-${name}`).forEach(cb => cb.checked = checked); }
    function checkParent(name, cid) { if(cid==='treeBroken') return; const p = document.querySelector(`#${cid}-grp-${name} .grp-cb`); const all = document.querySelectorAll(`.${cid}-item-${name}`); p.checked = Array.from(all).every(c => c.checked); }
    function resetTree(cid) { document.querySelectorAll(`#${cid} input`).forEach(cb => cb.checked = false); }

    function fillCode() {
       const n = document.getElementById('impName').value;
       const t = _data.importList.find(x => x.name === n);
       if(t) { document.getElementById('impCode').value = t.code||''; document.getElementById('impLimit').value = t.limit||5; }
    }

    // --- QUICK REPORT ---
    function quickReport(id) {
      switchTab('tabReport');
      _currentSpecificID = id;
      document.getElementById('groupFilter').style.display = 'none';
      document.getElementById('specificFilter').style.display = 'block';
      document.getElementById('lblSpecificID').innerText = id;
      msg('rptMsg', 'success', `Đang tìm ngày sử dụng...`);
      google.script.run.withSuccessHandler(r => {
        if(r.found) {
          document.getElementById('rptStart').value = r.start;
          document.getElementById('rptEnd').value = r.end;
          msg('rptMsg', 'success', `Đã tìm thấy ngày. Bấm xuất báo cáo ngay.`);
        } else {
          msg('rptMsg', 'error', 'Không tìm thấy ngày trong Log.');
        }
      }).getToolDateRange(id);
    }

    function clearSpecific() {
      _currentSpecificID = null;
      document.getElementById('specificFilter').style.display = 'none';
      document.getElementById('groupFilter').style.display = 'block';
      msg('rptMsg', 'success', 'Đã quay lại chọn theo nhóm.');
    }

    // --- BUTTON ACTIONS ---
    function doImport() {
      const d = { name: document.getElementById('impName').value, code: document.getElementById('impCode').value, limit: document.getElementById('impLimit').value, qty: document.getElementById('impQty').value };
      msg('impMsg', 'success', 'Đang xử lý...');
      google.script.run.withSuccessHandler(r=>{ msg('impMsg', 'success', r.message); load(); }).coreImportTools(d);
    }

    function doReport() {
      const s = document.getElementById('rptStart').value;
      const e = document.getElementById('rptEnd').value;
      const checkedBoxes = document.querySelectorAll('#treeReport .item-cb:checked');
      const selectedIDs = Array.from(checkedBoxes).map(cb => cb.value);
      if(!s || !e) return msg('rptMsg', 'error', 'Vui lòng chọn ngày.');
      if(!_currentSpecificID && selectedIDs.length === 0) return msg('rptMsg', 'error', 'Vui lòng chọn ít nhất 1 vật tư.');
      msg('rptMsg', 'success', 'Đang tạo báo cáo...');
      document.getElementById('rptLinks').style.display = 'none';
      const filter = { startDate: s, endDate: e, selectedIDs: selectedIDs, specificID: _currentSpecificID };
      google.script.run.withSuccessHandler(r => {
        if(r.success) {
          document.getElementById('lnkPdf').href = r.pdfUrl;
          document.getElementById('lnkExcel').href = r.excelUrl;
          document.getElementById('rptLinks').style.display = 'block';
          msg('rptMsg', 'success', 'Xong! Dữ liệu sẽ tự cập nhật...');
          // Tự động load lại sau 2s để xóa thông báo nếu là báo cáo chi tiết
          setTimeout(load, 2000); 
        } else msg('rptMsg', 'error', r.message);
      }).coreGenerateReport(filter);
    }

    function doBroken() {
        const sel = document.querySelector('#treeBroken input[type="radio"]:checked');
        if(!sel) return msg('brkMsg','error','Vui lòng chọn vật tư.');
        const d = { toolID: sel.value, reporter: document.getElementById('brkRep').value, witness: document.getElementById('brkWit').value, related: document.getElementById('brkRelated').value, description: document.getElementById('brkDesc').value };
        if(!d.reporter) return msg('brkMsg','error','Thiếu người lập');
        if(!confirm('Xác nhận khóa vĩnh viễn?')) return;
        msg('brkMsg','success','Đang xử lý...');
        google.script.run.withSuccessHandler(r=>{
            if(r.success) {
                msg('brkMsg','success',r.message);
                if(r.docUrl) { document.getElementById('lnkDoc').href = r.docUrl; document.getElementById('lnkDoc').style.display = 'inline-block'; }
                setTimeout(load, 2000);
            } else msg('brkMsg','error',r.message);
        }).reportBrokenTool(d);
    }

    // Xóa thủ công
    function manualFinish(id) {
      if(!confirm("Xóa thông báo này?\n(Đồng nghĩa đã xử lý xong)")) return;
      
      // Xóa ngay lập tức trên giao diện để người dùng thấy nhanh
      const card = document.getElementById('card-' + id);
      if(card) card.style.display = 'none';
      
      msg('notMsg', 'success', 'Đang xử lý...');
      google.script.run.withSuccessHandler(r => {
        if(r.success) { 
            // Load lại để đồng bộ hoàn toàn
            load(); 
            msg('notMsg', 'success', 'Đã xóa.'); 
        }
        else {
            if(card) card.style.display = 'flex'; // Hiện lại nếu lỗi
            alert(r.message);
        }
      }).markToolHandled(id);
    }

    function switchTab(id) {
      document.querySelectorAll('.content').forEach(e=>e.classList.remove('active'));
      document.querySelectorAll('.tab').forEach(e=>e.classList.remove('active'));
      document.getElementById(id).classList.add('active');
      document.querySelector(`.tab[onclick="switchTab('${id}')"]`).classList.add('active');
    }

    function msg(id, type, txt) {
      const e = document.getElementById(id); e.className = 'message '+type; e.textContent = txt; e.style.display='block';
    }
